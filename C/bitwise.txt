//set a bit
#include<stdio.h>
void main()
{
    int i,j,k,num,pos;
    printf("enter any number:\n");
    scanf("%d",&num);
    printf("enter position to set:\n");
    scanf("%d",&pos);
    k=num|(1<<pos);
    printf("%d\n",k);
}



//clear a bit
#include<stdio.h>
void main()
{
    int i,j,k,num,pos;
    printf("enter any number:\n");
    scanf("%d",&num);
    printf("enter position to set:\n");
    scanf("%d",&pos);
    k=num&(~(1<<pos));
    printf("%d\n",k);
}


//toggle a bit
#include<stdio.h>
void main()
{
    int i,j,k,num,pos;
    printf("enter any number:\n");
    scanf("%d",&num);
    printf("enter position to toggle:\n");
    scanf("%d",&pos);
    k=num^(1<<pos);
    printf("%d\n",k);
}





//number of set bits

#include<stdio.h>
void main()
{
    int i,j,k,num;
    printf("enter any number:\n");
    scanf("%d",&num);
    for(i=0;i<32;i++)
    {
        if(num&(1<<i))
        {
            j++;
        }
    }
    printf("%d\n",j);
}





//number of clear bits

#include<stdio.h>
void main()
{
    int i,j=0,k=0,num;
    printf("enter any number:\n");
    scanf("%d",&num);
    for(i=0;i<32;i++)
    {
        if(num&(1<<i))
        {
            j++;
        }
        else
        {
            k++;
        }
    }
    printf("set bits:%d\n",j);
    printf("clear bits:%d\n",k);
}
