    printf("entered string is:\n");
    puts(a);
    printf("\n");
    for(i=0;a[i]!='\0';i++)
    {
        count++;
    }
    for(i=0;i<count;i++)
    {
        if(a[i]>='a'&&a[i]<='z')
        {

        else if(a[i]>='A'&&a[i]<='Z')
        {
            continue;
        }
    }
    printf("string after onverting in to uppercase is:\n");

//string reverse
#include<stdio.h>
void main()
{
    char a[34],i,j,k,count=0,count1=0,temp;
    printf("enter any string:\n");
    gets(a);
    printf("entered string is:\n");
    puts(a);
    for(i=0;a[i]!='\0';i++)
    {
        count++;
    }
    j=count-1;
    for(i=0;i<count/2;i++)
    {
        temp=a[i];
        a[i]=a[j];
        a[j]=temp;
        j--;
    }
    printf("\n");
    printf("reverse of given string is:\n");
    puts(a);
}
