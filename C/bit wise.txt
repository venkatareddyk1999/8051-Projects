/*Write a C program to check Least Significant Bit (LSB) of a number is set or 
not.*/
#include<stdio.h>
void main()
{
    int num;
    printf("enter any number:\n");
    scanf("%d",&num);
    if(num&1)
    {
        printf("LSB of given number is 1\n");
    }
    else
    {
        printf("LSB of given number is 0\n");
    }
}









/*Write a C program to check Most Significant Bit (MSB) of a number is set or 
not.*/
#include<stdio.h>
void main()
{
    int num,bits,msb;
    printf("enter any number:\n");
    scanf("%d",&num);
    bits=sizeof(num)*8;
    msb=(1<<(bits-1));
    if(msb&num)
    {
        printf("msb bit is 1\n");
    }
    else
    {
        printf("msb bit is 0\n");
    }
    
}





//Write a C program to get nth bit of a number.

#include<stdio.h>
void main()
{
    int num,n,status;
    printf("enter any number:\n");
    scanf("%d",&num);
    printf("enter bit position to check:\n");
    scanf("%d",&n);
    status=(num>>n)&1;
    if(status==1)
    {
        printf("given position is set\n");
    }
    else
    {
        printf("given position is clear\n");
    }
}

//get highest set bit of a number

#include<stdio.h>
void main()
{
    int num,count=-1,i=0,bits;
    printf("enter any number:\n");
    scanf("%d",&num);
    bits=sizeof(num)*8;
    for(i=0;i<bits;i++)
    {
        if((num>>i)&1)
        count=i;
    }
    if(count!=-1)
    printf("%d\n",count);
    else
    printf("0 has no set bits");
}










//lowest set bits of a number
#include<stdio.h>
void main()
{
    int i,num,count=-1,bits;
    printf("enter any number:\n");
    scanf("%d",&num);
    bits=sizeof(num)*8;
    for(i=bits;i>=0;i--)
    {
        if((num>>i)&1)
        {
            count=i;
        }

    if(count!=-1)
    {
        printf(" lowest set bit of a number is:%d\n",count);
    }
    else
    {

    }
}
