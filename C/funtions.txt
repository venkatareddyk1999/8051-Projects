//maximum and minimum between two numbers
#include<stdio.h>
int maxmin(int a,int b);
int minmax(int a,int b);
void main()
{
    int a,b,max,min;
    printf("enter any two numbers:");
    scanf("%d%d",&a,&b);
    max=maxmin(a,b);
    min=minmax(a,b);
    printf("maximum number is %d\n",max);
    printf("minimum number is:%d\n",min);
}
int maxmin(int a,int b)
{
    int max;
    max=(a>b)?a:b;
    return max;
}
int minmax(int a,int b)
{
    int max;
    max=(a<b)?a:b;
    return max;
}
